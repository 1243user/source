存储插件：store.js
富编辑器：xheditor.js
表格插件：bsgrid.js/datatable.js
字符统计: charcount.js
图片裁剪:jquery.cropper.min.js
分页插件：jquery.pagination.js
时间日历：jquery.timepicker.js
树型列表：jquery.ztree.min.js / jquery.treeview.js
预览插件：jquery.uploadPreview.js
